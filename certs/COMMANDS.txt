# Génération des certificats SSL pour les services
# Ce fichier contient les commandes à exécuter dans Git Bash ou WSL

# Distance Service
openssl genrsa -out distance-service.key 2048
openssl req -new -x509 -key distance-service.key -out distance-service.crt -days 365 -subj "/C=FR/ST=IDF/L=Paris/O=Hospital-POC/OU=Dev/CN=distance-service"
openssl pkcs12 -export -in distance-service.crt -inkey distance-service.key -out distance-service.p12 -name distance-service -passout pass:changeit
